### 3. The Builder Script (scripts/build_route_consistency_records.py)

This script reads your existing corpora, runs them through the live atomic_tokenization + chemical_fusion pipeline, assigns route metadata, groups by intent_id + target_cluster, and emits validated JSONL. It is ready to run today.

Python

Copy

```
#!/usr/bin/env python3
from __future__ import annotations

import json
import uuid
from pathlib import Path
from typing import Any

import numpy as np

from python.scbe.atomic_tokenization import AtomicTokenization  # your existing module
from python.scbe.chemical_fusion import ChemicalFusionPipeline     # your existing module

class RouteConsistencyBuilder:
    def __init__(self):
        self.atomic = AtomicTokenization()          # your live atomic mapper
        self.fusion = ChemicalFusionPipeline()      # your live fusion engine

    def build_record(self, raw_text: str, language: str, tokenizer_source: str,
                     tongue: str, layer: str, view: str) -> dict[str, Any]:
        record_id = str(uuid.uuid4())

        # 1. Atomic tokenization + STISA
        tokens, elements, features, trits = self.atomic.process(raw_text, language, tokenizer_source)

        # 2. Chemical fusion
        fusion_result = self.fusion.fuse(trits, features)

        # 3. Route metadata
        intent_id = f"intent_{hash(raw_text) % 1000000:06d}"   # deterministic for same text
        target_cluster = f"exec_vec_{hash(raw_text) % 100000:05d}"

        return {
            "record_id": record_id,
            "intent_id": intent_id,
            "route_id": f"{tongue.lower()}_{layer}_{view}",
            "route_family": "polyhedral_projection",
            "target_cluster": target_cluster,
            "source_corpus": "training-data/...",
            "input": {"text": raw_text, "language": language, "tokenizer_source": tokenizer_source},
            "route_metadata": {
                "tongue": tongue,
                "layer": layer,
                "view": view,
                "governance": "ALLOW",                     # placeholder — replace with real SCBE call
                "witness_stable": any(e.startswith("WIT") for e in elements)  # noble-gas style
            },
            "atomic_features": {
                "tokens": tokens,
                "elements": elements,
                "features": features.tolist(),
                "trits": trits.tolist()
            },
            "chemical_fusion": fusion_result,
            "triangulation_links": [],                     # populated later by grouping pass
            "labels": {"passes_governance": True, "stability_score": float(fusion_result["rhombic_score"])}
        }

    def build_from_corpus(self, input_dir: str | Path, output_jsonl: str | Path):
        input_dir = Path(input_dir)
        output_path = Path(output_jsonl)
        output_path.parent.mkdir(parents=True, exist_ok=True)

        records = []
        for file in input_dir.glob("*.jsonl"):
            for line in file.open(encoding="utf-8"):
                data = json.loads(line)
                raw_text = data.get("text") or data.get("prompt") or ""
                if not raw_text:
                    continue

                # Generate one record per route (example: 3 routes)
                for route in [("KO", "L3", "canonical"), ("AV", "L4", "trl_conversation"), ("RU", "L3", "governance_cls")]:
                    record = self.build_record(raw_text, data.get("language", "en"), "STISA", *route)
                    records.append(record)

        # Simple grouping pass for triangulation_links (same intent_id)
        from collections import defaultdict
        clusters = defaultdict(list)
        for r in records:
            clusters[r["intent_id"]].append(r["record_id"])

        for r in records:
            r["triangulation_links"] = [
                {"kind": "semantic_process_execution", "linked_record_ids": clusters[r["intent_id"]]}
            ]

        with output_path.open("w", encoding="utf-8") as f:
            for r in records:
                f.write(json.dumps(r, ensure_ascii=False) + "\n")

        print(f"✅ Built {len(records)} route-consistency records → {output_path}")

if __name__ == "__main__":
    builder = RouteConsistencyBuilder()
    builder.build_from_corpus("training-data/", "training-data/route_consistency_v1.jsonl")
```

```
#!/usr/bin/env python3
"""
SCBE-AETHERMOORE Route Consistency Record Builder (Optimized v2)
- Streaming JSONL input/output
- Batched + parallel route generation
- Deterministic hashing
- Schema validation
- Progress + stats
"""

from __future__ import annotations

import argparse
import hashlib
import json
import sys
from collections import defaultdict
from concurrent.futures import ProcessPoolExecutor, as_completed
from pathlib import Path
from typing import Any, Dict, List

import numpy as np
from tqdm import tqdm

# Import your live SCBE modules (no changes needed)
from python.scbe.atomic_tokenization import AtomicTokenization
from python.scbe.chemical_fusion import ChemicalFusionPipeline

# Optional: Pydantic for runtime schema validation
try:
    from pydantic import BaseModel, Field, ValidationError
except ImportError:
    BaseModel = None  # graceful fallback


class RouteConsistencyRecord(BaseModel):
    """Exact schema from our earlier design (Pydantic v2 style)."""
    record_id: str
    intent_id: str
    route_id: str
    route_family: str = Field(..., pattern=r"^(polyhedral_projection|tongue_view|layer_projection|contrast_pair)$")
    target_cluster: str
    source_corpus: str
    input: Dict[str, Any]
    route_metadata: Dict[str, Any]
    atomic_features: Dict[str, Any]
    chemical_fusion: Dict[str, Any]
    triangulation_links: List[Dict[str, Any]]
    labels: Dict[str, Any]


class OptimizedRouteConsistencyBuilder:
    def __init__(self, batch_size: int = 64, max_workers: int = None, validate: bool = True):
        self.batch_size = batch_size
        self.max_workers = max_workers or min(8, len(__import__('os').sched_getaffinity(0)) // 2 or 4)
        self.atomic = AtomicTokenization()
        self.fusion = ChemicalFusionPipeline()
        self.validate = validate and BaseModel is not None

    def _hash(self, text: str, route_id: str) -> str:
        """Deterministic 64-char hex hash (collision-resistant for our scale)."""
        key = f"{text.strip()}_{route_id}".encode("utf-8")
        return hashlib.sha256(key).hexdigest()

    def process_text(self, raw_text: str, language: str, tokenizer_source: str,
                     routes: List[tuple[str, str, str]]) -> List[Dict[str, Any]]:
        """Single-text, multi-route processor (called in parallel)."""
        records = []
        for tongue, layer, view in routes:
            route_id = f"{tongue.lower()}_{layer}_{view}"
            intent_id = self._hash(raw_text, "intent")
            target_cluster = self._hash(raw_text, "cluster")

            # 1. Atomic tokenization (your existing call)
            tokens, elements, features, trits = self.atomic.process(
                raw_text, language, tokenizer_source
            )

            # 2. Chemical fusion
            fusion_result = self.fusion.fuse(trits, features)

            record = {
                "record_id": self._hash(raw_text, route_id),
                "intent_id": intent_id,
                "route_id": route_id,
                "route_family": "polyhedral_projection",
                "target_cluster": target_cluster,
                "source_corpus": "training-data",
                "input": {"text": raw_text[:2000], "language": language, "tokenizer_source": tokenizer_source},  # truncate for safety
                "route_metadata": {
                    "tongue": tongue,
                    "layer": layer,
                    "view": view,
                    "governance": "ALLOW",  # ← replace with live /v1/authorize call later
                    "witness_stable": any(e.startswith(("WIT", "NOBLE")) for e in elements)
                },
                "atomic_features": {
                    "tokens": tokens,
                    "elements": elements,
                    "features": features.tolist() if isinstance(features, np.ndarray) else features,
                    "trits": trits.tolist() if isinstance(trits, np.ndarray) else trits
                },
                "chemical_fusion": fusion_result,
                "triangulation_links": [],  # populated in grouping pass
                "labels": {
                    "passes_governance": True,
                    "stability_score": float(fusion_result.get("rhombic_score", 0.0))
                }
            }

            if self.validate:
                try:
                    RouteConsistencyRecord(**record)
                except ValidationError as e:
                    print(f"Validation error on {route_id}: {e}", file=sys.stderr)
                    continue

            records.append(record)
        return records

    def build_from_corpus(self, input_dir: str | Path, output_jsonl: str | Path):
        input_dir = Path(input_dir)
        output_path = Path(output_jsonl)
        output_path.parent.mkdir(parents=True, exist_ok=True)

        # Fixed set of routes (easy to extend)
        routes = [
            ("KO", "L3", "canonical"),
            ("AV", "L4", "trl_conversation"),
            ("RU", "L3", "governance_cls"),
            # Add more polyhedral routes here as needed
        ]

        all_records = []
        batch = []
        stats = defaultdict(int)

        # Streaming read
        for jsonl_file in tqdm(list(input_dir.glob("*.jsonl")), desc="Scanning corpora"):
            with jsonl_file.open(encoding="utf-8") as f:
                for line_num, line in enumerate(f, 1):
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        data = json.loads(line)
                        raw_text = data.get("text") or data.get("prompt") or data.get("input", "")
                        if not raw_text or len(raw_text) < 5:
                            continue
                        batch.append((raw_text, data.get("language", "en"), data.get("tokenizer_source", "STISA")))

                        if len(batch) >= self.batch_size:
                            # Parallel process batch
                            with ProcessPoolExecutor(max_workers=self.max_workers) as executor:
                                futures = [executor.submit(self.process_text, text, lang, src, routes) for text, lang, src in batch]
                                for future in as_completed(futures):
                                    all_records.extend(future.result())
                            batch.clear()

                    except (json.JSONDecodeError, KeyError) as e:
                        print(f"Skipping malformed line {line_num} in {jsonl_file}: {e}", file=sys.stderr)

        # Final batch
        if batch:
            with ProcessPoolExecutor(max_workers=self.max_workers) as executor:
                futures = [executor.submit(self.process_text, text, lang, src, routes) for text, lang, src in batch]
                for future in as_completed(futures):
                    all_records.extend(future.result())

        # Efficient grouping pass for triangulation_links
        clusters = defaultdict(list)
        for r in all_records:
            clusters[r["intent_id"]].append(r["record_id"])

        for r in all_records:
            r["triangulation_links"] = [{
                "kind": "semantic_process_execution",
                "linked_record_ids": clusters[r["intent_id"]]
            }]

        # Streaming write
        with output_path.open("w", encoding="utf-8") as f:
            for record in tqdm(all_records, desc="Writing JSONL"):
                f.write(json.dumps(record, ensure_ascii=False) + "\n")
                stats[record["route_metadata"]["tongue"]] += 1

        # Final stats
        total = len(all_records)
        print(f"\n✅ Built {total:,} route-consistency records → {output_path}")
        print("Route distribution:")
        for tongue, count in sorted(stats.items()):
            print(f"  {tongue:>2}: {count:>8,} ({count/total*100:5.1f}%)")
        print(f"  Witness-stable fraction: {sum(1 for r in all_records if r['route_metadata'].get('witness_stable')) / total:.1%}")


def main():
    parser = argparse.ArgumentParser(description="Optimized SCBE Route Consistency Record Builder")
    parser.add_argument("--input-dir", default="training-data", help="Directory of source JSONL corpora")
    parser.add_argument("--output", default="training-data/route_consistency_v2.jsonl", help="Output JSONL path")
    parser.add_argument("--batch-size", type=int, default=128, help="Batch size for atomic/fusion calls")
    parser.add_argument("--workers", type=int, default=None, help="Max parallel workers (default: auto)")
    parser.add_argument("--no-validate", action="store_true", help="Skip Pydantic schema validation")
    args = parser.parse_args()

    builder = OptimizedRouteConsistencyBuilder(
        batch_size=args.batch_size,
        max_workers=args.workers,
        validate=not args.no_validate
    )
    builder.build_from_corpus(args.input_dir, args.output)


if __name__ == "__main__":
    main()
```
