import numpy as np  
  
def rhombic_fusion(x, A, V, G, k,  
                   alpha=1.0, beta=1.0, gamma=1.0, eta=0.5, phi=1.618):  
  
    uA = np.abs(A(x))  
    uV = np.abs(V(x))  
    g  = G(x)  
  
    E01 = np.linalg.norm(x - uA)**2  
    E02 = np.linalg.norm(x - uV)**2  
    E13 = np.linalg.norm(uA - g)**2  
    E23 = np.linalg.norm(uV - g)**2  
    E12 = np.linalg.norm(uA - uV)**2  
  
    phase = (-1/phi)**(k % 3)  
  
    R = (alpha*(E01+E02) +  
         beta*(E13+E23) +  
         gamma*E12 +  
         eta*phase*E12)  
  
    return R
